syntax for if stm:
===================
 if(condition)
 {
    //block
 }

 syntax for if-else stm:
=======================
 if(condition)
 {
    //if block(true)
 }
 else
 {
    //else-block(false)
 }

syntax for if else if statement
===============================

if(condition)
{
    //if block
}
else if (condition)
{
    //else if block
}
else if (condition)
{
    //else if block
}
else if (condition)
{
    //else if block
}
else{

    //else block
}

===============================
